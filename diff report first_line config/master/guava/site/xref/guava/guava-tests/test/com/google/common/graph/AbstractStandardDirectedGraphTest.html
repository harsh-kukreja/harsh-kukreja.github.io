312   /**
313    * Populates the graph with nodes and edges in a star shape with node `1` in the middle.
314    *
315    * <p>Note that the edges are added in a shuffled order to properly test the effect of the
316    * insertion order.
317    */
318   private void populateStarShapedGraph() {
319     putEdge(2, 1);
320     putEdge(1, 4);
321     putEdge(1, 3);
322     putEdge(5, 1);
323     putEdge(1, 2);
324     putEdge(3, 1);
325   }

362   /**
363    * Tests that the method {@code putEdge} will silently add the missing nodes to the graph, then
364    * add the edge connecting them. We are not using the proxy methods here as we want to test {@code
365    * putEdge} when the end-points are not elements of the graph.
366    */
367   @Test
368   public void putEdge_nodesNotInGraph() {
369     assume().that(graphIsMutable()).isTrue();
370 
371     graphAsMutableGraph.addNode(N1);
372     assertTrue(graphAsMutableGraph.putEdge(N1, N5));
373     assertTrue(graphAsMutableGraph.putEdge(N4, N1));
374     assertTrue(graphAsMutableGraph.putEdge(N2, N3));
375     assertThat(graph.nodes()).containsExactly(N1, N5, N4, N2, N3).inOrder();
376     assertThat(graph.successors(N1)).containsExactly(N5);
377     assertThat(graph.successors(N2)).containsExactly(N3);
378     assertThat(graph.successors(N3)).isEmpty();
379     assertThat(graph.successors(N4)).containsExactly(N1);
380     assertThat(graph.successors(N5)).isEmpty();
381   }
