15  package com.google.common.util.concurrent.internal;
16  
17  /**
18   * Static utilities for {@link InternalFutureFailureAccess}. Most users will never need to use this
19   * class.
20   *
21   * <p>This class is GWT-compatible.
22   *
23   * @since {@code com.google.guava:failureaccess:1.0}, which was added as a dependency of Guava in
24   *     Guava 27.0
25   */
26  public final class InternalFutures {
27    /**
28     * Usually returns {@code null} but, if the given {@code Future} has failed, may <i>optionally</i>
29     * return the cause of the failure. "Failure" means specifically "completed with an exception"; it
30     * does not include "was cancelled." To be explicit: If this method returns a non-null value,
31     * then:
32     *
33     * <ul>
34     *   <li>{@code isDone()} must return {@code true}
35     *   <li>{@code isCancelled()} must return {@code false}
36     *   <li>{@code get()} must not block, and it must throw an {@code ExecutionException} with the
37     *       return value of this method as its cause
38     * </ul>
39     */
40    public static Throwable tryInternalFastPathGetFailure(InternalFutureFailureAccess future) {
41      return future.tryInternalFastPathGetFailure();
42    }
43  
44    private InternalFutures() {}
45  }
